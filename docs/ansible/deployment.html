Ansible Playbook Guide

This guide covers advanced playbook techniques, deployment strategies, and production-ready automation patterns for Ansible.

Advanced Playbook Patterns

Multi-Environment Deployments

Environment-Based Configuration

Use variable files and group_vars to manage different environments.

Directory Structure:
├── inventories/
│   ├── development/
│   │   ├── hosts.ini
│   │   └── group_vars/
│   │       ├── all.yml
│   │       └── webservers.yml
│   ├── staging/
│   │   ├── hosts.ini
│   │   └── group_vars/
│   │       ├── all.yml
│   │       └── webservers.yml
│   └── production/
│       ├── hosts.ini
│       └── group_vars/
│           ├── all.yml
│           └── webservers.yml
├── playbooks/
│   └── deploy-webapp.yml
└── roles/
    ├── nginx/
    ├── app/
    └── database/
Environment Variables Example:
# group_vars/all.yml (shared across environments)
---
project_name: mywebapp
app_user: deploy

# inventories/development/group_vars/all.yml
---
env: development
debug_mode: true
database_host: dev-db.internal
app_instances: 1

# inventories/production/group_vars/all.yml
---
env: production
debug_mode: false
database_host: prod-db.internal
app_instances: 3
Deployment Command:
# Deploy to different environments
ansible-playbook -i inventories/development playbooks/deploy-webapp.yml
ansible-playbook -i inventories/production playbooks/deploy-webapp.yml

Blue-Green Deployments

Zero-Downtime Application Updates

Deploy new version alongside old version, then switch traffic.

Blue-Green Playbook:
---
- name: Blue-Green Deployment
  hosts: load_balancers
  vars:
    blue_version: "1.0.0"
    green_version: "1.1.0"
    active_color: "{{ lookup('file', '/tmp/active_color') | default('blue') }}"
  tasks:
    - name: Determine target color
      set_fact:
        target_color: "{{ 'green' if active_color == 'blue' else 'blue' }}"

    - name: Deploy to target environment
      include_role:
        name: deploy_app
      vars:
        app_color: "{{ target_color }}"
        app_version: "{{ green_version if target_color == 'green' else blue_version }}"

    - name: Run health checks
      uri:
        url: "http://{{ target_color }}-app.example.com/health"
        status_code: 200
      register: health_check
      until: health_check.status == 200
      retries: 30
      delay: 10

    - name: Switch load balancer
      include_role:
        name: update_loadbalancer
      vars:
        active_color: "{{ target_color }}"

    - name: Update active color file
      copy:
        content: "{{ target_color }}"
        dest: /tmp/active_color
      delegate_to: localhost

    - name: Scale down old environment
      include_role:
        name: scale_app
      vars:
        app_color: "{{ active_color }}"
        desired_instances: 0

Canary Deployments

Risk-Reduced Rollouts

Gradually roll out changes to a subset of users/servers.

Canary Deployment Strategy:
---
- name: Canary Deployment
  hosts: webservers
  serial: 1  # Deploy to one server at a time
  max_fail_percentage: 20

  pre_tasks:
    - name: Check current app version
      slurp:
        src: /opt/app/version.txt
      register: current_version

  tasks:
    - name: Deploy to canary server
      include_role:
        name: deploy_app
      vars:
        app_version: "{{ new_version }}"
      when: inventory_hostname in groups['canary']

    - name: Health check on canary
      uri:
        url: "http://{{ inventory_hostname }}/health"
        status_code: 200
      when: inventory_hostname in groups['canary']

    - name: Wait for manual approval
      pause:
        prompt: "Canary deployment successful. Continue with full rollout? (yes/no)"
      when: inventory_hostname in groups['canary'] | first
      run_once: true
      register: approval

    - name: Fail if not approved
      fail:
        msg: "Deployment cancelled by user"
      when: approval.user_input | default('no') != 'yes'

    - name: Full rollout
      include_role:
        name: deploy_app
      vars:
        app_version: "{{ new_version }}"
      when: inventory_hostname not in groups['canary']

Role-Based Architecture

Creating Reusable Roles

Modular Role Design

Build roles that are composable and reusable across projects.

Role Structure Best Practices:
roles/
├── nginx/
│   ├── defaults/
│   │   └── main.yml          # Default variables (lowest precedence)
│   ├── vars/
│   │   └── main.yml          # Internal role variables
│   ├── tasks/
│   │   ├── main.yml          # Main task file
│   │   ├── install.yml       # Installation tasks
│   │   └── configure.yml     # Configuration tasks
│   ├── templates/
│   │   ├── nginx.conf.j2     # Configuration templates
│   │   └── vhost.conf.j2     # Virtual host templates
│   ├── files/
│   │   └── ssl/              # Static files
│   ├── handlers/
│   │   └── main.yml          # Service handlers
│   ├── meta/
│   │   └── main.yml          # Role metadata and dependencies
│   └── README.md             # Documentation
Role Dependencies (meta/main.yml):
---
dependencies:
  - role: common
    tags: ['common']
  - role: ssl
    when: nginx_ssl_enabled | default(false)
    tags: ['ssl']
  - role: monitoring
    when: monitoring_enabled | default(true)
    tags: ['monitoring']

Role Composition

Combining Multiple Roles

Use role composition to build complex applications from simple components.

Multi-Role Playbook:
---
- name: Deploy Web Application Stack
  hosts: webservers
  become: yes

  vars:
    app_name: mywebapp
    app_version: "1.2.3"
    environment: production

  pre_tasks:
    - name: Update package cache
      package:
        update_cache: yes
      when: ansible_os_family == "Debian"
      tags: ['setup']

  roles:
    - role: common
      tags: ['common']

    - role: nginx
      nginx_ssl_enabled: true
      nginx_vhosts:
        - name: "{{ app_name }}"
          server_name: "{{ app_name }}.example.com"
          ssl_cert: "/etc/ssl/certs/{{ app_name }}.pem"
          ssl_key: "/etc/ssl/private/{{ app_name }}.key"
      tags: ['nginx']

    - role: php
      php_version: "8.1"
      php_modules:
        - php-mysql
        - php-curl
        - php-gd
      tags: ['php']

    - role: app
      app_name: "{{ app_name }}"
      app_version: "{{ app_version }}"
      app_environment: "{{ environment }}"
      tags: ['app']

    - role: monitoring
      monitoring_targets:
        - "{{ app_name }}"
      when: environment == "production"
      tags: ['monitoring']

  post_tasks:
    - name: Verify deployment
      uri:
        url: "http://localhost/health"
        status_code: 200
      tags: ['verify']

Testing Strategies

Automated Testing with Molecule

Molecule Configuration:
# molecule/default/molecule.yml
---
dependency:
  name: galaxy
  options:
    role-file: requirements.yml
driver:
  name: docker
platforms:
  - name: instance
    image: ubuntu:20.04
    privileged: true
provisioner:
  name: ansible
  inventory:
    host_vars:
      instance:
        ansible_user: ubuntu
verifier:
  name: ansible
Converge Playbook:
# molecule/default/converge.yml
---
- name: Converge
  hosts: all
  become: yes
  gather_facts: yes

  roles:
    - role: nginx
      nginx_port: 8080
      nginx_server_name: localhost

  post_tasks:
    - name: Test nginx service
      uri:
        url: http://localhost:8080
        status_code: 200
Running Tests:
# Run full test cycle
molecule test

# Run individual steps
molecule create    # Create test instances
molecule converge  # Run playbook
molecule verify    # Run tests
molecule destroy   # Clean up

Security Best Practices

Secrets Management

Secure Credential Handling

Ansible Vault:
# Create encrypted file
ansible-vault create secrets.yml

# Encrypt existing file
ansible-vault encrypt secrets.yml

# Edit encrypted file
ansible-vault edit secrets.yml

# View encrypted file
ansible-vault view secrets.yml

# Use vault password file
ansible-playbook --vault-password-file ~/.vault_pass deploy.yml
Vault Variables:
# group_vars/production/vault.yml (encrypted)
---
vault_mysql_root_password: verysecretpassword
vault_app_secret_key: anothersecretkey
vault_ssl_private_key: |
  -----BEGIN PRIVATE KEY-----
  MIIEvgIBADANBgkqhkiG9w0BAQEFAASCBKgwggSkAgEAAoIBAQC...
  -----END PRIVATE KEY-----
Using Vault Variables:
---
- name: Deploy application
  hosts: webservers
  vars_files:
    - vars/main.yml
    - vault.yml  # Encrypted file

  tasks:
    - name: Configure database
      template:
        src: config.php.j2
        dest: /var/www/config.php
      vars:
        db_password: "{{ vault_mysql_root_password }}"
        app_key: "{{ vault_app_secret_key }}"

SSH Security

Secure SSH Configuration

SSH Config for Ansible:
# ~/.ssh/config
Host *.production.example.com
    User ansible
    IdentityFile ~/.ssh/production_key
    IdentitiesOnly yes
    StrictHostKeyChecking no
    UserKnownHostsFile /dev/null
    LogLevel ERROR

Host *.staging.example.com
    User ansible
    IdentityFile ~/.ssh/staging_key
    ControlMaster auto
    ControlPath ~/.ssh/master-%r@%h:%p
    ControlPersist 5m
Ansible SSH Configuration:
# ansible.cfg
[ssh_connection]
ssh_args = -o ControlMaster=auto -o ControlPersist=60s -o StrictHostKeyChecking=no
control_path = ~/.ssh/ansible-%%r@%%h:%%p
pipelining = True
scp_if_ssh = True

Performance Optimization

Parallel Execution

Optimizing Playbook Speed

Forks and Serial:
# ansible.cfg - Increase parallelism
[defaults]
forks = 50

# Playbook with controlled parallelism
---
- name: Fast deployment
  hosts: webservers
  serial: "30%"  # Deploy to 30% of hosts at once
  max_fail_percentage: 10

- name: Database migration
  hosts: databases
  serial: 1     # One at a time for safety

- name: Load balancer update
  hosts: loadbalancers
  serial: "{{ groups['loadbalancers'] | length }}"  # All at once
Async Tasks:
tasks:
  - name: Start long-running task
    command: /opt/app/upgrade.sh
    async: 3600  # Maximum runtime in seconds
    poll: 10     # Check every 10 seconds

  - name: Check async task status
    async_status:
      jid: "{{ ansible_job_id }}"
    register: job_result
    until: job_result.finished
    retries: 30
    delay: 10

Fact Gathering Optimization

Reducing Fact Gathering Overhead

Selective Fact Gathering:
---
- name: Minimal fact gathering
  hosts: webservers
  gather_facts: no  # Disable automatic fact gathering

  tasks:
    - name: Gather only network facts
      setup:
        gather_subset:
          - network
          - hardware

    - name: Gather specific facts
      setup:
        filter: ansible_*

- name: Use cached facts
  hosts: webservers
  gather_facts: yes
  # Facts cached for 24 hours by default

Error Handling and Rollback

Graceful Failure Handling

Robust Error Recovery

Block with Rescue:
tasks:
  - name: Attempt deployment
    block:
      - name: Stop application
        service:
          name: myapp
          state: stopped

      - name: Deploy new version
        include_role:
          name: deploy_app

      - name: Start application
        service:
          name: myapp
          state: started

    rescue:
      - name: Log failure
        debug:
          msg: "Deployment failed, starting rollback"

      - name: Rollback to previous version
        include_role:
          name: rollback_app

      - name: Start application
        service:
          name: myapp
          state: started

    always:
      - name: Send notification
        include_role:
          name: notify_team

Rollback Strategies

Automated Rollback Procedures

Version-Based Rollback:
---
- name: Application Rollback
  hosts: webservers
  vars:
    current_version: "{{ lookup('file', '/opt/app/current_version.txt') }}"
    rollback_version: "{{ lookup('file', '/opt/app/previous_version.txt') }}"

  tasks:
    - name: Stop application
      service:
        name: myapp
        state: stopped

    - name: Backup current state
      copy:
        src: "/opt/app/releases/{{ current_version }}/"
        dest: "/opt/app/backups/{{ ansible_date_time.iso8601 }}/"

    - name: Switch to previous version
      file:
        src: "/opt/app/releases/{{ rollback_version }}"
        dest: /opt/app/current
        state: link

    - name: Update version file
      copy:
        content: "{{ rollback_version }}"
        dest: /opt/app/current_version.txt

    - name: Start application
      service:
        name: myapp
        state: started

    - name: Verify rollback
      uri:
        url: "http://localhost/health"
        status_code: 200

Monitoring and Alerting

Deployment Monitoring

Track Deployment Success and Performance

Deployment Metrics:
---
- name: Monitor Deployment
  hosts: webservers
  tasks:
    - name: Record deployment start
      copy:
        content: |
          deployment_id: {{ deployment_id }}
          start_time: {{ ansible_date_time.iso8601 }}
          version: {{ app_version }}
          environment: {{ environment }}
        dest: /tmp/deployment_start.json
      delegate_to: localhost

    - name: Track task execution time
      include_tasks: track_task.yml
      loop: "{{ deployment_tasks }}"
      vars:
        task_start: "{{ ansible_date_time.iso8601 }}"

    - name: Send metrics to monitoring
      uri:
        url: "{{ monitoring_endpoint }}"
        method: POST
        body_format: json
        body:
          deployment_id: "{{ deployment_id }}"
          status: "{{ 'success' if ansible_playbook_success else 'failure' }}"
          duration: "{{ (ansible_date_time.epoch | int) - (deployment_start_time | int) }}"
          version: "{{ app_version }}"
      delegate_to: localhost
      run_once: true
