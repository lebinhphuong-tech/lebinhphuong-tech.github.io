kubectl Commands Cheatsheet

A comprehensive reference guide for kubectl commands. Bookmark this page for quick access to the most commonly used Kubernetes commands.

ðŸ’¡ Tip: Use kubectl --help or kubectl <command> --help for detailed help on any command.

Cluster Information

Basic Cluster Commands

Command Description
kubectl cluster-info Display cluster information
kubectl cluster-info dump Dump cluster state to stdout
kubectl version Show client and server version
kubectl version --client Show only client version
kubectl api-versions List all API versions supported
kubectl api-resources List all API resources

Nodes

Node Commands

# List all nodes
kubectl get nodes

# List nodes with more details
kubectl get nodes -o wide

# Describe a specific node
kubectl describe node <node-name>

# Get node metrics
kubectl top node <node-name>

# Get all node metrics
kubectl top nodes

# Cordon a node (mark as unschedulable)
kubectl cordon <node-name>

# Uncordon a node
kubectl uncordon <node-name>

# Drain a node (evict all pods)
kubectl drain <node-name>

# Drain with ignore daemonsets
kubectl drain <node-name> --ignore-daemonsets

# Label a node
kubectl label nodes <node-name> disktype=ssd

# Annotate a node
kubectl annotate nodes <node-name> description="Production node"

Pods

Pod Management

# List pods in current namespace
kubectl get pods

# List pods in all namespaces
kubectl get pods -A

# List pods with more details
kubectl get pods -o wide

# List pods in specific namespace
kubectl get pods -n <namespace>

# List pods with labels
kubectl get pods -l app=nginx

# List pods with custom output format
kubectl get pods -o yaml
kubectl get pods -o json
kubectl get pods -o name

# Describe a pod
kubectl describe pod <pod-name>

# Describe pods with label selector
kubectl describe pods -l app=nginx

# Get pod logs
kubectl logs <pod-name>

# Follow logs
kubectl logs -f <pod-name>

# Get logs from specific container
kubectl logs <pod-name> -c <container-name>

# Get logs from previous container
kubectl logs <pod-name> --previous

# Get logs with timestamp
kubectl logs <pod-name> --timestamps

# Get logs since specific time
kubectl logs <pod-name> --since=1h
kubectl logs <pod-name> --since=30m

# Get last N lines of logs
kubectl logs <pod-name> --tail=100

# Get logs from all pods with label
kubectl logs -l app=nginx --tail=100

# Execute command in pod
kubectl exec -it <pod-name> -- sh
kubectl exec <pod-name> -- ls /app

# Execute command in specific container
kubectl exec -it <pod-name> -c <container-name> -- sh

# Copy file from pod
kubectl cp <pod-name>:/path/to/file ./local-file

# Copy file to pod
kubectl cp ./local-file <pod-name>:/path/to/file

# Port forward to pod
kubectl port-forward <pod-name> 8080:80

# Delete pod
kubectl delete pod <pod-name>

# Delete pods with label
kubectl delete pods -l app=nginx

# Force delete pod
kubectl delete pod <pod-name> --grace-period=0 --force

Deployments

Deployment Operations

# List deployments
kubectl get deployments
kubectl get deploy

# Get deployment details
kubectl describe deployment <deployment-name>

# Create deployment
kubectl create deployment nginx --image=nginx:latest

# Create deployment with replicas
kubectl create deployment nginx --image=nginx:latest --replicas=3

# Apply deployment from YAML
kubectl apply -f deployment.yaml

# Update deployment
kubectl apply -f deployment.yaml

# Replace deployment
kubectl replace -f deployment.yaml

# Update deployment image
kubectl set image deployment/<deployment-name> nginx=nginx:1.21

# Scale deployment
kubectl scale deployment/<deployment-name> --replicas=5

# Get rollout status
kubectl rollout status deployment/<deployment-name>

# View rollout history
kubectl rollout history deployment/<deployment-name>

# View specific revision
kubectl rollout history deployment/<deployment-name> --revision=2

# Rollback to previous version
kubectl rollout undo deployment/<deployment-name>

# Rollback to specific revision
kubectl rollout undo deployment/<deployment-name> --to-revision=2

# Pause rollout
kubectl rollout pause deployment/<deployment-name>

# Resume rollout
kubectl rollout resume deployment/<deployment-name>

# Restart deployment
kubectl rollout restart deployment/<deployment-name>

# Delete deployment
kubectl delete deployment <deployment-name>

# Delete deployment but keep pods
kubectl delete deployment <deployment-name> --cascade=orphan

Services

Service Management

# List services
kubectl get services
kubectl get svc

# Get service details
kubectl describe service <service-name>

# Create service from YAML
kubectl apply -f service.yaml

# Expose deployment as service
kubectl expose deployment nginx --port=80 --type=LoadBalancer

# Expose with NodePort
kubectl expose deployment nginx --port=80 --type=NodePort

# Port forward to service
kubectl port-forward service/<service-name> 8080:80

# Get service endpoints
kubectl get endpoints <service-name>

# Delete service
kubectl delete service <service-name>

ConfigMaps & Secrets

ConfigMap Commands

# List ConfigMaps
kubectl get configmaps
kubectl get cm

# Describe ConfigMap
kubectl describe configmap <cm-name>

# Get ConfigMap YAML
kubectl get configmap <cm-name> -o yaml

# Create ConfigMap from literal
kubectl create configmap <cm-name> --from-literal=key1=value1 --from-literal=key2=value2

# Create ConfigMap from file
kubectl create configmap <cm-name> --from-file=config.properties

# Create ConfigMap from directory
kubectl create configmap <cm-name> --from-file=<directory>

# Create ConfigMap from YAML
kubectl apply -f configmap.yaml

# Edit ConfigMap
kubectl edit configmap <cm-name>

# Delete ConfigMap
kubectl delete configmap <cm-name>

Secret Commands

# List secrets
kubectl get secrets

# Describe secret
kubectl describe secret <secret-name>

# Get secret YAML
kubectl get secret <secret-name> -o yaml

# Create secret from literal
kubectl create secret generic <secret-name> --from-literal=username=admin --from-literal=password=secret

# Create secret from file
kubectl create secret generic <secret-name> --from-file=username.txt --from-file=password.txt

# Create secret for docker registry
kubectl create secret docker-registry <secret-name> --docker-server=<registry> --docker-username=<user> --docker-password=<pass>

# Create TLS secret
kubectl create secret tls <secret-name> --cert=tls.crt --key=tls.key

# Edit secret
kubectl edit secret <secret-name>

# Delete secret
kubectl delete secret <secret-name>

Namespaces

Namespace Operations

# List namespaces
kubectl get namespaces
kubectl get ns

# Create namespace
kubectl create namespace <namespace-name>

# Describe namespace
kubectl describe namespace <namespace-name>

# Get all resources in namespace
kubectl get all -n <namespace>

# Set default namespace for current context
kubectl config set-context --current --namespace=<namespace-name>

# View current namespace
kubectl config view --minify | grep namespace

# Delete namespace (deletes all resources)
kubectl delete namespace <namespace-name>

# Label namespace
kubectl label namespace <namespace-name> environment=production

Resource Management

Common Resource Commands

# Get all resources
kubectl get all

# Get all resources in namespace
kubectl get all -n <namespace>

# Get specific resource type
kubectl get <resource-type>

# Get with labels
kubectl get pods -l app=nginx

# Get with multiple labels
kubectl get pods -l app=nginx,environment=production

# Watch resources
kubectl get pods -w

# Get resources sorted
kubectl get pods --sort-by=.metadata.creationTimestamp

# Get resources with custom columns
kubectl get pods -o custom-columns=NAME:.metadata.name,STATUS:.status.phase,NODE:.spec.nodeName

# Edit resource
kubectl edit deployment <deployment-name>

# Delete resource
kubectl delete deployment <deployment-name>

# Delete multiple resources
kubectl delete deployment nginx redis postgres

# Delete by label
kubectl delete pods -l app=nginx

# Delete all resources in namespace
kubectl delete all --all -n <namespace>

Scaling & Autoscaling

Manual & Auto Scaling

# Scale deployment manually
kubectl scale deployment/<deployment-name> --replicas=5

# Scale with current replicas check
kubectl scale deployment/<deployment-name> --replicas=5 --current-replicas=3

# Create Horizontal Pod Autoscaler
kubectl autoscale deployment/<deployment-name> --min=2 --max=10 --cpu-percent=70

# Get HPA
kubectl get hpa

# Describe HPA
kubectl describe hpa <hpa-name>

# Delete HPA
kubectl delete hpa <hpa-name>

# View pod metrics
kubectl top pods

# View node metrics
kubectl top nodes

# View metrics for specific pod
kubectl top pod <pod-name>

Context & Configuration

Kubectl Configuration

# View current context
kubectl config current-context

# List all contexts
kubectl config get-contexts

# Switch context
kubectl config use-context <context-name>

# View configuration
kubectl config view

# View specific context
kubectl config view --context=<context-name>

# Set context
kubectl config set-context <context-name> --cluster=<cluster-name> --user=<user-name> --namespace=<namespace>

# Set default namespace for context
kubectl config set-context --current --namespace=<namespace-name>

# Rename context
kubectl config rename-context <old-name> <new-name>

# Delete context
kubectl config delete-context <context-name>

# Get clusters
kubectl config get-clusters

# Get users
kubectl config get-users

Debugging & Troubleshooting

Debug Commands

# Get events
kubectl get events

# Get events sorted by time
kubectl get events --sort-by='.lastTimestamp'

# Get events in namespace
kubectl get events -n <namespace>

# Watch events
kubectl get events --watch

# Describe resource (detailed info)
kubectl describe pod <pod-name>

# Get resource YAML
kubectl get pod <pod-name> -o yaml

# Get resource JSON
kubectl get pod <pod-name> -o json

# Get resource with JSONPath
kubectl get pod <pod-name> -o jsonpath='{.status.phase}'

# Dry run (preview changes)
kubectl apply -f deployment.yaml --dry-run=client

# Validate YAML
kubectl apply -f deployment.yaml --validate=true --dry-run=client

# Diff between live and file
kubectl diff -f deployment.yaml

# Execute into pod for debugging
kubectl exec -it <pod-name> -- /bin/sh

# Run temporary pod for debugging
kubectl run debug-pod --image=busybox -it --rm -- sh

Advanced Operations

Advanced Commands

# Apply multiple YAML files
kubectl apply -f deployment.yaml -f service.yaml

# Apply all YAML files in directory
kubectl apply -f <directory>/

# Patch resource
kubectl patch deployment <deployment-name> -p '{"spec":{"replicas":5}}'

# Label resource
kubectl label pod <pod-name> environment=production

# Remove label
kubectl label pod <pod-name> environment-

# Annotate resource
kubectl annotate pod <pod-name> description="Production pod"

# Remove annotation
kubectl annotate pod <pod-name> description-

# Drain node (evict pods)
kubectl drain <node-name> --ignore-daemonsets --delete-emptydir-data

# Cordon node (mark unschedulable)
kubectl cordon <node-name>

# Uncordon node
kubectl uncordon <node-name>

# Get API versions
kubectl api-versions

# Get API resources
kubectl api-resources

# Explain resource
kubectl explain pod
kubectl explain pod.spec
kubectl explain pod.spec.containers

Output Formats

Flag Output Format

-o yaml YAML format
-o json JSON format
-o name Resource names only
-o custom-columns= Custom column format
-o jsonpath= JSONPath expression
-o go-template= Go template
